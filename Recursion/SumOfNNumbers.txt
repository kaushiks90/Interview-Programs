using System;

namespace ConsoleApp
{
    internal class Program
    {
        public static void Main(string[] args)
        {
            Console.WriteLine(Addition(10));
            Console.ReadLine();
        }

        public static int Addition(int n)
        {
            if (n == 0)
            {
                return 0;
            }
            int sum=Addition(n - 1);
            sum+= n;
            return sum;
        }

    }
   
}
